function get_post_data(post_url) {

    const output = {}

    const [channel_name, server_name] = document.querySelector(`head > title`).textContent.split(` | `)

    const [server_id, channel_id, post_id] = post_url.match(/\d+/g)

    const post_message = document.querySelector(`#message-content-${post_id}`).textContent.trim()
    const post_timestamp = document.querySelector(`#message-timestamp-${post_id}`).ariaLabel
    const post_element = document.querySelector(`#chat-messages-${channel_id}-${post_id}`).outerHTML

    const key = `${server_id}_${channel_id}_${post_id}`

    output[key] = {

        "server_id": server_id,
        "server_name": server_name,
        "channel_id": channel_id,
        "channel_name": channel_name,
        "post_url": post_url,
        "post_id": post_id,
        "post_message": post_message,
        "post_timestamp": post_timestamp,
        "post_element": post_element
    }

    output_json_pretty = JSON.stringify(output, null, 4)

    console.log(output_json_pretty)

}
